Risk Assessment Algorithm


import numpy as np
import pandas as pd
from sklearn.ensemble import RandomForestClassifier

# Credit Risk Scoring Model
def calculate_risk_score(customer_data):
    """Calculate comprehensive risk score for loan approval"""
    features = ['income', 'credit_history', 'debt_ratio', 'employment_years']
    
    # Apply weighted scoring algorithm
    risk_weights = {
        'income': 0.3,
        'credit_history': 0.4,
        'debt_ratio': 0.2,
        'employment_years': 0.1
    }
    
    score = sum(customer_data[feature] * weight 
                for feature, weight in risk_weights.items())
    
    return min(max(score, 0), 100)  # Normalize to 0-100 scale
