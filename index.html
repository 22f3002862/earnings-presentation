def calculate_comprehensive_risk_score(customer_data):
    """Calculate risk score for loan approval decisions"""
    key_features = ['income', 'credit_history', 'debt_ratio', 'employment_years']
    
    # Apply machine learning weighted scoring
    risk_weights = {
        'income': 0.30,
        'credit_history': 0.40, 
        'debt_ratio': 0.20,
        'employment_years': 0.10
    }
    
    # Calculate weighted score
    weighted_score = sum(customer_data[feature] * weight 
                        for feature, weight in risk_weights.items())
    
    # Normalize to 0-100 scale
    final_score = min(max(weighted_score, 0), 100)
    return final_score
